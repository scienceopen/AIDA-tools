function P = atm_att_est(Intens,zenith_a) 
% ATM_ATT_EST  Estimate of the atmospheric absorption 
% from variation of the star intensity with zenith angle 
% 
 
% Copyright Bjorn Gustavsson 20030901 
 
I = []; 
Z = []; 
for i = 1:size(Intens,1), 
 
  [in] = find(Intens); 
  I = [I, Intens(in)]; 
  Z = [Z, zenith_a(in)]; 
 
end 
 
P = polyfit(1./cos(Z),log(I),1);
